4. db_helpers.py — Internal Logic

%%{init: {'theme':'dark', 'themeVariables': {'fontSize':'13px'}}}%%
flowchart TD
    subgraph QM["query_movies(db, query, page, limit)"]
        QM1["Get movies collection"] --> QM2["collection.find(query or {})"]
        QM2 --> QM3{"page > 0?"}
        QM3 -->|Yes| QM4["cursor.skip · limit"]
        QM3 -->|No| QM5["Return all results"]
        QM4 --> QM6["Convert _id → string"]
        QM5 --> QM6
        QM6 --> QM7["Return jsonify(movies)"]
    end

    subgraph PO["parse_object_id(id_string)"]
        PO1["Try ObjectId(id_string)"] --> PO2{"Valid?"}
        PO2 -->|Yes| PO3["Return ObjectId"]
        PO2 -->|No / InvalidId| PO4["abort(400)"]
    end

    subgraph RJ["require_json(request)"]
        RJ1{"request.is_json?"} -->|No| RJ2["abort(400)
Content-Type error"]
        RJ1 -->|Yes| RJ3["get_json(silent=True)"]
        RJ3 --> RJ4{"body is None?"}
        RJ4 -->|Yes| RJ5["abort(400)
Invalid JSON"]
        RJ4 -->|No| RJ6["Return parsed body"]
    end
